CREATE TABLE user(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR (100) NOT NULL,
email VARCHAR (100) NOT NULL,
phone VARCHAR (100) NOT NULL,
city VARCHAR (100) NOT NULL);

CREATE TABLE technology(
techid INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR (100) NOT NULL,
isActive INT  NOT NULL);

CREATE TABLE user_tech(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
userid INT  NOT NULL,
techid INT  NOT NULL);


ALTER TABLE `user_tech`
    ADD CONSTRAINT `ut_user` FOREIGN KEY (`userid`) REFERENCES `user`(`id`) ON DELETE CASCADE,
    ADD CONSTRAINT `ut_tech` FOREIGN KEY (`techid`) REFERENCES `technology`(`techid`) ON DELETE CASCADE;
